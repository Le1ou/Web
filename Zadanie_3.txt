Написать функцию, принимающую на вход int, и число, обратное этому int

def ReverseValue(value):
    result = 0
    value_copy = value
    while(value != 0):
        if(value < 0):
            digit = (0-value) % 10
        else:
            digit = value % 10
        result = result*10 + digit
        value = int(value/10)
    if(value_copy < 0):
        return (0 - result)
    else:
        return result
value = int(input("Введите число: "))
